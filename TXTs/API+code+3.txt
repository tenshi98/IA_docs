import openai

openai.api_key = "XXXXXXXXXXXXX"

messages = []
system_msg = input("What custom instructions would you like to implement\n")
messages.append({"role": "system", "content": system_msg})

print("Ok, your assistant's custom instructions are now set up. How can I help you?")
while input != "exit!":
    message = input()
    messages.append({"role": "user", "content": message})
    response = openai.chat.completions.create(
        model="gpt-3.5-turbo",
        messages=messages)
    reply = response.choices[0].message.content
    messages.append({"role": "assistant", "content": reply})
    print("\n" + reply + "\n")
